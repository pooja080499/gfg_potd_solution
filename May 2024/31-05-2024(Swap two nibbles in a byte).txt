class Solution {
    static int swapNibbles(int n) {
        String binary = Integer.toBinaryString(n);
         while(binary.length()<8)
            {
                binary='0'+binary;
            }
        String newbinary=binary.substring(4)+binary.substring(0,4);
        int decimal=Integer.parseInt(newbinary,2); 
        return decimal;
    }
}
